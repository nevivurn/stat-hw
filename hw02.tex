\subsection*{Problem 2}

\begin{align*}
	P(\textrm{all not faulty})
	&= P(\textrm{1st not faulty} \cap \textrm{2nd not faulty} \cap \textrm{3rd not faulty}) \\
	&= P(\textrm{1st not faulty}) P(\textrm{2nd not faulty}) P(\textrm{3rd not faulty}) \\
	&= (1-0.05)^3 \\
	&= 0.857
\end{align*}

\subsection*{Problem 10}

\begin{itemize}
	\item \textbf{(a)}: $P(X \ge 12.6) = \int_{12.6}^{\infty} 20e^{-20(x-12.5)}\,dx =
		\int_{0.1}^{\infty} 20e^{-20x}\,dx = e^{-2} = 0.135$
	\item \textbf{(b)}:
		\begin{itemize}
			\item $E(X) = \int_{12.5}^{\infty} = 20xe^{-20(x-12.5)}\,dx =

			\item $E(X^2) =\int_{12.5}^{\infty} = 20x^{2}e^{-20(x-12.5)}\,dx =

			\item $\mathrm{Var}(X) = E(X^2) - \mu^2 = 0.0025$
		\end{itemize}
\end{itemize}
